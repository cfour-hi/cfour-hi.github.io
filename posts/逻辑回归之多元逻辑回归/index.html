def predict(X, theta):
    m, n = X.shape
    X = np.hstack((np.ones((m, 1)), X))
    h = softmax(np.dot(X, theta))
    return np.argmax(h, axis=1)

def accuracy(y_pred, y_true):
    return np.mean(y_pred == y_true) * 100
